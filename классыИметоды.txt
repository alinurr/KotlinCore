fun calculateDiscountedPrice(
    price: Double,
    discountPercent: Int = 10,
    applyVat: Boolean = true
): Double {
    if (price <= 0) {
        println("‚ùå –ù–µ–¥–æ–ø—É—Å—Ç–∏–º–∞—è —Ü–µ–Ω–∞: $price")
        return 0.0
    }

    if (discountPercent <= 0) {
        println("‚ö†Ô∏è –°–∫–∏–¥–∫–∞ –Ω–µ –ø—Ä–∏–º–µ–Ω—è–µ—Ç—Å—è.")
        val basePrice = if (applyVat) price * 1.2 else price
        return basePrice
    }

    if (discountPercent >= 100) {
        println("üéâ –ë–µ—Å–ø–ª–∞—Ç–Ω–æ! –°–∫–∏–¥–∫–∞ 100% –∏–ª–∏ –±–æ–ª—å—à–µ.")
        return 0.0
    }

    val discount = price * (discountPercent / 100.0)
    var finalPrice = price - discount

    if (applyVat) {
        finalPrice *= 1.2
    }


//–≤–æ–∑–≤—Ä–∞—â–µ–Ω–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è –∏–∑ —Ñ—É–Ω–∫—Ü–∏–∏
    return finalPrice
}
